#include <stdio.h>
#include <stdlib.h>

#define MAX 100

// Huffman Tree Node
struct Node {
    char data;
    int freq;
    struct Node *left, *right;
};

// Create new node
struct Node* newNode(char data, int freq) {
    struct Node* temp = (struct Node*)malloc(sizeof(struct Node));
    temp->data = data;
    temp->freq = freq;
    temp->left = temp->right = NULL;
    return temp;
}

// Simple priority queue using array (min frequency first)
void sortNodes(struct Node* arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j]->freq > arr[j + 1]->freq) {
                struct Node* temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

// Print Huffman Codes
void printCodes(struct Node* root, int arr[], int top) {
    if (root->left) {
        arr[top] = 0;
        printCodes(root->left, arr, top + 1);
    }

    if (root->right) {
        arr[top] = 1;
        printCodes(root->right, arr, top + 1);
    }

    // If leaf node
    if (!root->left && !root->right) {
        printf("%c: ", root->data);
        for (int i = 0; i < top; i++)
            printf("%d", arr[i]);
        printf("\n");
    }
}

int main() {
    int n = 6;

    char data[] = {'a', 'b', 'c', 'd', 'e', 'f'};
    int freq[]  = {5, 9, 12, 13, 16, 45};

    struct Node* nodes[10];

    for (int i = 0; i < n; i++) {
        nodes[i] = newNode(data[i], freq[i]);
    }

    // Build Huffman Tree
    while (n > 1) {
        sortNodes(nodes, n);

        struct Node* left = nodes[0];
        struct Node* right = nodes[1];

        struct Node* parent = newNode('$', left->freq + right->freq);
        parent->left = left;
        parent->right = right;

        // Replace first two nodes with new parent
        nodes[0] = parent;
        for (int i = 1; i < n - 1; i++) {
            nodes[i] = nodes[i + 1];
        }

        n--;
    }

    int arr[MAX], top = 0;

    printf("Huffman Codes:\n");
    printCodes(nodes[0], arr, top);

    return 0;
}
