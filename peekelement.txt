#include <stdio.h>

// Function to find a peak element using divide and conquer
int findPeak(int arr[], int low, int high, int n) {
    int mid = low + (high - low) / 2;

    // Check if mid is peak
    if ((mid == 0 || arr[mid] >= arr[mid - 1]) &&
        (mid == n - 1 || arr[mid] >= arr[mid + 1])) {
        return arr[mid];
    }

    // If left neighbor is greater, then peak is in left half
    if (mid > 0 && arr[mid - 1] > arr[mid]) {
        return findPeak(arr, low, mid - 1, n);
    }

    // Else peak is in right half
    return findPeak(arr, mid + 1, high, n);
}

int main() {
    int arr[] = {1, 3, 20, 4, 1, 0};
    int n = sizeof(arr) / sizeof(arr[0]);

    int peak = findPeak(arr, 0, n - 1, n);

    printf("Peak Element is: %d\n", peak);

    return 0;
}
